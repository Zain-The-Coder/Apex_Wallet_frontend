<script>
    /**
     * Password Visibility Toggle
     */
    function togglePass(id) {
        const input = document.getElementById(id);
        const btn = input.nextElementSibling.querySelector('span');
        if (input.type === 'password') {
            input.type = 'text';
            btn.innerText = 'visibility_off';
        } else {
            input.type = 'password';
            btn.innerText = 'visibility';
        }
    }

    /**
     * View Switching Logic
     */
    function switchView(view) {
        const login = document.getElementById('loginView');
        const register = document.getElementById('registerView');
        hideError();

        if (view === 'register') {
            login.classList.add('opacity-0', '-translate-x-8');
            setTimeout(() => {
                login.classList.add('hidden');
                register.classList.remove('hidden');
                setTimeout(() => {
                    register.classList.remove('opacity-0', 'translate-x-8');
                    register.classList.add('opacity-100', 'translate-x-0');
                }, 50);
            }, 400);
        } else {
            register.classList.add('opacity-0', 'translate-x-8');
            setTimeout(() => {
                register.classList.add('hidden');
                login.classList.remove('hidden');
                setTimeout(() => {
                    login.classList.remove('opacity-0', '-translate-x-8');
                    login.classList.add('opacity-100', 'translate-x-0');
                }, 50);
            }, 400);
        }
    }

    /**
     * Email Format Validation
     */
    function validateEmail(input) {
        const feedback = document.getElementById('emailFeedback');
        const re = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
        
        if (re.test(input.value)) {
            feedback.classList.remove('hidden');
            input.classList.remove('border-error');
            input.classList.add('border-secondary');
        } else {
            feedback.classList.add('hidden');
            input.classList.remove('border-secondary');
            if (input.value.length > 0) {
                input.classList.add('border-error');
            } else {
                input.classList.remove('border-error');
            }
        }
    }

    /**
     * Password Strength Meter Logic
     */
    function checkStrength(val) {
        const bars = [document.getElementById('bar1'), document.getElementById('bar2'), document.getElementById('bar3'), document.getElementById('bar4')];
        const text = document.getElementById('strengthText');
        
        // Reset styles
        bars.forEach(b => b.className = 'h-full w-1/4 bg-outline-variant transition-colors');
        
        let score = 0;
        if (val.length >= 6) score++;
        if (val.length >= 10) score++;
        if (/[A-Z]/.test(val) && /[0-9]/.test(val)) score++;
        if (/[^A-Za-z0-9]/.test(val)) score++;

        const colors = ['bg-error', 'bg-tertiary', 'bg-primary-container', 'bg-secondary'];
        const labels = ['Weak', 'Medium', 'Strong', 'Secure'];

        if (val.length === 0) {
            text.innerText = 'Weak';
            text.className = 'font-label-sm text-label-sm text-outline';
            return;
        }

        for (let i = 0; i < score; i++) {
            bars[i].className = `h-full w-1/4 ${colors[score-1]} transition-colors`;
        }
        
        text.innerText = labels[score-1] || 'Weak';
        
        // Visual feedback for text color
        const colorClassMap = ['text-error', 'text-tertiary', 'text-primary', 'text-secondary'];
        text.className = `font-label-sm text-label-sm ${colorClassMap[score-1] || 'text-outline'}`;
    }

    /**
     * Error Banner Management
     */
    function hideError() {
        document.getElementById('errorAlert').classList.add('hidden');
    }

    /**
     * Form Submission Handler
     */
    function handleAuth(e, type) {
        e.preventDefault();
        const btn = e.target.querySelector('button[type="submit"]');
        const text = btn.querySelector('.btn-text');
        const loader = btn.querySelector('.btn-loader');
        const alert = document.getElementById('errorAlert');
        const errorMsg = document.getElementById('errorText');

        // Toggle Loading State
        btn.disabled = true;
        text.classList.add('hidden');
        loader.classList.remove('hidden');
        loader.classList.add('flex');

        // Simulate Network Delay
        setTimeout(() => {
            // Demonstration: "error@apex.com" always fails
            const emailInput = e.target.querySelector('input[type="email"]');
            
            if (emailInput && emailInput.value === 'error@apex.com') {
                // Fail state simulation
                btn.disabled = false;
                text.classList.remove('hidden');
                loader.classList.add('hidden');
                loader.classList.remove('flex');
                
                errorMsg.innerText = type === 'login' 
                    ? 'Incorrect email or password. Please try again.' 
                    : 'This email is already registered.';
                alert.classList.remove('hidden');
            } else {
                // Success state simulation
                showSuccess();
            }
        }, 2000);
    }

    /**
     * Final Success Transition
     */
    function showSuccess() {
        const overlay = document.getElementById('successOverlay');
        const circle = document.getElementById('successCircle');
        
        overlay.classList.remove('pointer-events-none');
        overlay.classList.add('opacity-100');
        
        setTimeout(() => {
            circle.classList.remove('scale-50');
            circle.classList.add('scale-100');
            
            // In a real app, window.location would change here
            console.log("Redirecting...");
        }, 100);
    }
